    实验五基于UDP的编程

    实验要求

    一实验目的:

    熟练掌握UDP下的网络编程

    二、实验内容：

    1、回射程序的基本功能：即回声，客户端发送什么数据，服务器端就客户端返回什么数据，服务器并显示客户端的IP地址即端口号。服务器端口号为4000.

    2、计算面积功能：

    服务器端口号为4000，客户端将矩形的长和宽发给服务器，服务器计算出矩形的面积，并把面积发送给客户端，客户端收到后，显示出来。

    实验1

    ntohs()    			网络字节位转主机字节位
inet_ntoa()			将网络地址转换成“.”点隔的字符串格式。

    服务器代码：

    // 实验服务端.cpp : Defines the entry point for the console application.
//

#include "stdafx.h"
#include<winsock2.h>
int main(int argc, char* argv[])
{
    WSADATA data;
    
    WSAStartup(MAKEWORD(2,2),&data);
    SOCKET s=socket(AF_INET,SOCK_DGRAM,IPPROTO_UDP);
    if(s==INVALID_SOCKET)
    {
        printf("套接字初始化失败!\n");
        return -1;
    }
    struct sockaddr_in hint;
    hint.sin_port=htons(4000);
    hint.sin_family=AF_INET;
    hint.sin_addr.S_un.S_addr=inet_addr("172.22.60.78");
    bind(s,(const struct sockaddr *)&hint,sizeof(hint));
    printf("服务器创建成功！\n");
    char buf[100]={0};
    struct sockaddr_in addr;
    int len=sizeof(addr);
    while(1){
    memset(buf,0,sizeof(buf));
    int i=recvfrom(s,buf,100,0,(struct sockaddr *)&addr,&len);
    if(i>=0)
    {
        printf("接收成功，数据为：%s,来自IP=%s,prot=%d\n",buf,inet_ntoa(addr.sin_addr),ntohs(addr.sin_port));
    }
    else
    {
        printf("接收失败\n");
    }
    }
    WSACleanup();

}

#include<winsock2.h>
int main(int argc, char* argv[])
{
    WSADATA data;
    
    WSAStartup(MAKEWORD(2,2),&data);
    SOCKET s=socket(AF_INET,SOCK_DGRAM,IPPROTO_UDP);
    if(s==INVALID_SOCKET)
    {
        printf("套接字初始化失败!\n");
        return -1;
    }
    struct sockaddr_in hint;
    hint.sin_port=htons(4000);
    hint.sin_family=AF_INET;
    hint.sin_addr.S_un.S_addr=inet_addr("172.22.60.78");
    bind(s,(const struct sockaddr *)&hint,sizeof(hint));
    printf("服务器创建成功！\n");
    char buf[100]={0};
    struct sockaddr_in addr;
    int len=sizeof(addr);
    int a,b;
    
    int i=recvfrom(s,buf,100,0,(struct sockaddr *)&addr,&len);
    a=*(int *)buf;
    b=*(int *)(buf+sizeof(int));
    if(i>=0)
    {
        printf("接收到长：%d,宽：%d，来自IP=%s,prot=%d\n",a,b,inet_ntoa(addr.sin_addr),ntohs(addr.sin_port));
    }
    else
    {
        printf("接收失败\n");
    }
    int c=a*b;
    memset(buf,0,sizeof(buf));
    *(int *)buf=c;
    int j=sendto(s,buf,sizeof(int),0,(struct sockaddr *)&addr,len);
    

    WSACleanup();

}
